Real Estate Search

http://www.immobilienscout24.de/Suche/S-3/Wohnung-Kauf/{{state}}/{{city}}

This endpoint returns the prices of the most expensive real estate objects in Berlin, Germany. The city is a parameter so that the endpoint can be used for any city.→

Code
prices = $('.result-list-entry__primary-criterion:first-child').get() .map(function(elements){ return elements.textContent.trim().split('€')[0] }); locations = $('.result-list-entry__address').get() .map(function(elements){ return elements.textContent.trim().split('€')[0] }); var results = [] for (var i = 0; i < prices.length ; i++) { results.push({ price: prices[i] + '€', location: locations[i] || '' }) } return results

Flight departure times

http://www.berlin-airport.de/en/travellers-txl/arrivals-and-departures/departures/

This endpoint goes to the berlin-airport.de website and extracts the next flights that depart from the SXF airport.→

Code
results = [] var destinations = $('tr:not(:first-child) .flightAirport') var times = $('tr:not(:first-child) .flightDeptArr') destinations.each(function(id, element){ var item = { 'destination': element.innerHTML, 'time': times[id].innerHTML.trim() } results.push(item) }) return results

CasperJS Endpoints

ebay Search

http://www.ebay.com/

This endpoint goes to the ebay.com website and searches for the search term 'Drone'.→

Code
var result = {}; casper.start('http://www.ebay.com/'); casper.then(function() { casper.sendKeys("#gh-ac", extracty.args.search); }) casper.then(function() { casper.click("#gh-btn"); }) casper.waitForSelector(".vip", null, function() { callback('Failed waiting for ".vip"'); }); casper.then(function() { var value = casper.evaluate(function() { var texts = Array.prototype.slice.call(document.querySelectorAll(".vip")) .map(function(e) { return e.textContent.trim(); }); return texts; }) result.title = value; }) casper.then(function() { var value = casper.evaluate(function() { var texts = Array.prototype.slice.call(document.querySelectorAll(".vip")) .map(function(e) { return e.href; }); return texts; }) result.link = value; }) casper.waitForSelector(".prc .bold", null, function() { callback('Failed waiting for ".prc .bold"'); }); casper.then(function() { var value = casper.evaluate(function() { var texts = Array.prototype.slice.call(document.querySelectorAll(".prc .bold")) .map(function(e) { return e.textContent.trim().replace(/\s.*/g, ''); }); return texts; }) result.price = value; }) var zip = (result) => { var keys = Object.keys(result); return result[keys[0]].map((_, i) => { var x = {}; keys.forEach(e => x[e] = result[e][i]); return x }); } casper.run(function() { callback(zip(result)); });

Airbnb Search

This performs an Airbnb search and returns the listing results.→

Code
var result = {}; casper.start(`https://www.airbnb.de/s/${extracty.args.location}?guests=${extracty.args.guests}&adults=${extracty.args.adults}&checkin=${extracty.args.checkin}&checkout=${extracty.args.checkout}`); casper.waitForSelector(".listing-name--display", null, function() { callback('Failed waiting for ".listing-name--display"'); }); casper.then(function() { result = casper.evaluate(function() { var result = {}; result.name = Array.prototype.slice.call(document.querySelectorAll(".listing-name--display")) .map(function(e) { return e.textContent.trim(); }); result.guests = Array.prototype.slice.call(document.querySelectorAll(".hide-md span+ span")) .map(function(e) { return e.textContent.trim().replace(/ .*/, ''); }); result.link = Array.prototype.slice.call(document.querySelectorAll(".search-results div.panel-body.panel-card-section h3 > a")) .map(function(e) { return e.href; }); result.image = Array.prototype.slice.call(document.querySelectorAll(".img-responsive-height")) .map(function(e) { return e.src; }); result.price = Array.prototype.slice.call(document.querySelectorAll(".listing .price-amount")) .map(function(e) { return e.textContent.trim(); }); return result; }) }) var zip = (result) => { var keys = Object.keys(result); return result[keys[0]].map((_, i) => { var x = {}; keys.forEach(e => x[e] = result[e][i]); return x }); } casper.run(function() { callback(zip(result)); });
